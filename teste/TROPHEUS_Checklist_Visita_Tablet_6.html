✅ Como funciona
Cada visita salva é enviada para o Google Sheets automaticamente. Funciona de qualquer dispositivo. 100% gratuito.
📋 Configurar Google Apps Script (5 min)
1. Crie uma planilha em sheets.google.com
2. Menu: Extensões → Apps Script
3. Cole o script abaixo e salve
4. Implantar → Nova implantação → App da Web → Qualquer pessoa
5. Copie a URL e cole no campo acima
function doPost(e){
  var sheet=SpreadsheetApp.getActiveSpreadsheet()
    .getSheetByName("Visitas")||
    SpreadsheetApp.getActiveSpreadsheet()
    .insertSheet("Visitas");
  var visits=JSON.parse(e.postData.contents);
  if(!Array.isArray(visits))visits=[visits];
  if(sheet.getLastRow()===0){
    sheet.appendRow(["ID","Obra","Data","Empresa",
    "Responsavel","Endereco","Pavimentos","Area",
    "Etapa","Status","Oportunidade","Urgencia",
    "Servicos","Proximos Passos","Prazo Proposta",
    "Resp Interno","Obs Comercial","Obs Eletrica",
    "Potencial","Salvo Em"]);
  }
  visits.forEach(function(v){
    var row=[v.id,v.obra_nome,v.visita_data,
    v.empresa,v.resp_nome,v.obra_endereco,
    v.num_pav,v.area_m2,v.etapa_atual,
    v.status_obra,v.opp_grau,v.urgencia,
    v.servicos,v.next_steps,v.prazo_prop,
    v.resp_int,v.obs_com,v.obs_ele,
    v.potencial,v.savedAt];
    sheet.appendRow(row);
  });
  return ContentService
    .createTextOutput(JSON.stringify({ok:true}))
    .setMimeType(ContentService.MimeType.JSON);
}
function doGet(e){
  return ContentService
    .createTextOutput(JSON.stringify({ok:true}))
    .setMimeType(ContentService.MimeType.JSON);
}
